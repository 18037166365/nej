// 首先生成播放对象,只负责逻辑部分，要配合UI来使用
// preload：是否预加载
// url：音频地址
var _mda = _p._$$MediaAudio._$allocate({
    preload:false,
    url:'http://127.0.0.1:8000/nej-baseline/res/test.mp3',
    onstatechange:function(_event){
        // 状态改变的回调
        // 0 | 当前停止状态
        // 1 | 当前缓冲状态
        // 2 | 当前播放状态
        // 3 | 当前暂停状态
    }
});
// 开始播放
_mda._$play();
// 暂停播放
_mda._$pause();
// 停止播放
_mda._$stop();
