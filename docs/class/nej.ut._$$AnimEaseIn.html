var _box = document.getElementById('id-bounce1'),_easein;
var options = {
    from:{
        offset: 100,
        velocity: 10
    },
    to:{
        offset:200
    },
    duration:1000,
    onupdate: function(offset){
        _box.style.left = offset.offset + 'px';
    },
    onstop: function(){
        _easein = nej.ut._$$AnimEaseIn._$recycle(_easein);
    }
}
// 创建减速动画实例
_easein  = nej.ut._$$AnimEaseIn._$allocate(options);
// 开始动画
_easein._$play();
