var e = NEJ.P('nej.e');
var _box = e._$get('pagebox');
var _pg = e._$page(_box,{
    event:'click',
    index:9,
    total:10,
    onchange:function(_obj){
        // 返回页码信息，last:上一次页面,index:当前页面，total:总页数
    }
});
// 翻到第4页
_pg._$setIndex(4);
// 改变总页数,状态复原到第一页
_pg._$getTotal(20);
// 想要保持页面信息状态
_pg._$updatePage(5,10);
