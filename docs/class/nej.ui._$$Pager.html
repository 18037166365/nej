var p = NEJ.P('nej.ui'),
    v = NEJ.P('nej.v'),
    t = NEJ.P('nej.ut'),
    e = NEJ.P('nej.e');
// 默认第一页
var _setIndex = 1;
// 页面更改的回调方法
var _onchangeHandle = function(_obj){
    var _index = _obj.index;
};
// 实例化一个pager对象，总共10页
var _pager = p._$$Pager._$allocate({
    parent:'pagerCnt',
    onchange: _onchangeHandle,
    total: 10,
    index:_setIndex
});
// 从第2页翻到第10页
for(var i = 2 ; i < 11 ; i++){
    _setIndex = i;
    _pager._$setIndex(_setIndex);
}
// 绑定一个翻页器,视觉上翻页器会联动，但最后触发一次翻页器的回调,避免重复触发
_pager._$bind('pagerCnt2');
