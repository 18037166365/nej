    function(_event){
        // 必须先验证消息来源_event.origin是否你允许的域
         if (!_isAllow(_event.origin))
            return;
             
        // 处理_event.data中的消息内容
        // TODO something
         
        // 回复消息，使用_event.source
        nej.j._$postMessage(_event.source,{

            origin:_event.origin
        });
    });

• 输入

类型：String|Window
描述：window对象或者Frame的name，或者字符串如_top、_parent、_self
类型：Object
描述：消息配置
名称类型描述
dataVariable消息内容
originString目标Origin，只有指定的页面可以收到消息，如http://a.b.com
sourceString当前窗体标识，除非你非常确定当前窗体的标识是什么，否则请采用自动识别
