var _u = NEJ.P(nej.u),
     _list = [1,2,3],
     _list2 = [4,5,6];
 // 合并列表，把列表2加到列表1的起始位子，列表1从2开始截断
 // 生成结果[4,5,6,2,3]
 _u._$mergeList(_list,_list2,{union:true,
   begin:true,compare:function(_item){
       return _item == '2';
   }
});
