var _v = NEJ.P('nej.v');
 
// 事件回调中阻止链接默认事件
_v._$addEvent(
    'b','click',
    function(_event){
        // 阻止链接打开页面的默认行为
        _v._$stopDefault(_event);
    },false);
 
// a节点上的点击事件仍然会触发
_v._$addEvent(
    'a','click',
    function(_event){
        alert(0);
        // TODO something
    },false);
