var _v = NEJ.P('nej.v');
 
// 添加事件
_v._$addEvent(
  'abc','mouseover',
  function(_event){
      // TODO something
  },false);
 
_v._$addEvent(
  'abc','mouseover',
  function(_event){
      // TODO something
  },true);
 
_v._$addEvent(
  'abc','custom',
  function(_event){
      // TODO something
  },false);
 
// 清除节点所有事件，包括两个mouseover事件和一个custom事件
_v._$clearEvent('abc');
 
// 清除节点指定类型事件，只清除两个mouseover事件
_v._$clearEvent('abc','mouseover');
