var _obj0 = {a:0,b:1},
    _obj1 = {a:"a",b:"b",c:"c"};
var _filter = function(_value,_key){
    if (_key != "a")
        return false;
}
// 根据_obj1的属性拷贝对象到_obj0,过滤器过滤掉所有key值不是a的属性
// 结果是_obj0.a = 0,_obj.b = "b",_obj.c = "c";
var _obj = NEJ.X(_obj0,_obj1,_filter);
