• 简介

触发对象的某个事件，注：对于IE浏览器该接口有以下限制
  • 捕获阶段支持需要浏览器IE9+
  • 自定义事件支持需要浏览器IE9+

var _v = NEJ.P('nej.v');
 
// 注册鼠标事件
_v._$addEvent('abc','click',function(_event){
    // 获取鼠标事件触发的垂直位置
   var _y = _v._$pageY(_event);
},false);
// 触发鼠标事件
_v._$dispatchEvent('abc','click');
 
// 注册自定义事件
_v._$addEvent('abc','ok',function(_event){
   // TODO something
},false);
// 触发自定义事件
_v._$dispatchEvent('abc','ok');
